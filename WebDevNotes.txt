JS:

Function .getElementByClassName() seems useful
- Seems to be the way to do for loops and stuff 


<!-- What this is doing ->
<!--  -->

<script> 
  var accord_elems = document.getElementsByClassName("accordian");
  var i;

  for (i = 0; i < accord_elems.length; i++) {
    accord_elems[i].addEventListener("click", function() {
      this.classList.toggle("active");
      var panel = this.nextElementSibling;
      if(panel.style.display === "block") {
        panel.style.display = "none";
      } else {
        panel.style.display = "block";
      }
    })
  }
</script>
